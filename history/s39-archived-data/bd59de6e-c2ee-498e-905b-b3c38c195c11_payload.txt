    32	## Project Summary
    33	
    34	### Original Problem
    35	LRN sells recruiting service packages to help athletes get noticed by college coaches. The old system used **Wix Pricing Plans** which only offered popup checkout, no BNPL support, and no promo codes.
    36	
    37	### Solution
    38	Migrated to **Wix Stores** using a "fake physical product" workaround (physical products with free shipping to unlock subscription functionality). This gives us:
    39	- ✅ Full checkout page (not popup)
    40	- ✅ BNPL support (Afterpay, Klarna confirmed working)
    41	- ✅ Promo code / coupon functionality
    42	- ✅ Email checkout links (send checkout to parent while on a sales call)
    43	- ✅ Custom HTML pricing table with payment term selector
    44	- ✅ Member CMS auto-update on purchase
    45	- ✅ Page gating for subscribers
    46	- ✅ Order notifications with plan name
    47	- ✅ Additional Services add-ons (added Feb 14, 2026)
    48	
    49	### The Two Checkout Flows
    50	
    51	**Flow 1: Direct Checkout (Select Plan)**
    52	```
    53	You select plan + payment term + optional add-ons → Click "Select Plan" → Products added to YOUR cart → Navigate to /cart-page → Checkout
    54	```
    55	
    56	**Flow 2: Email Checkout (Send Plan)**
    57	```
    58	You select plan + payment term + optional add-ons → Click "Send Plan" → Enter parent email → Click "Email Checkout Link"
    59	→ Backend creates checkout template → Saves PendingCheckouts record → Sends branded HTML email via Google Apps Script
    60	→ Parent clicks "Complete Selection" in email → Guest checkout on Wix → onOrderPaid fires
    61	→ events.js matches PendingCheckouts record → Updates ATHLETE's Members record (not buyer's)
    62	```

  function startTracking(checkoutTemplateId, athleteName, planName, sendToEmail) {
    currentCheckoutTemplateId = checkoutTemplateId;

  function resetStatusSteps() {

    document.getElementById("timePurchased").textContent = "";

  function updateStatusTracker(data) {
    if (!data.found) return;

    var lineFill = 0;
    
    document.getElementById("stepSent").className = "status-step completed";

    if (data.emailOpenedAt) {
      document.getElementById("stepOpened").className = "status-step completed";
      document.getElementById("timeOpened").textContent = formatTime(new Date(data.emailOpenedAt));
      lineFill = 33;
      
      if (!data.linkClickedAt) {
        document.getElementById("stepClicked").className = "status-step active";
      }
    } else {
      document.getElementById("stepOpened").className = "status-step active";
    }

    if (data.linkClickedAt) {
      document.getElementById("stepClicked").className = "status-step completed";
      document.getElementById("timeClicked").textContent = formatTime(new Date(data.linkClickedAt));
      lineFill = 66;
      
      if (data.status !== "completed") {
        document.getElementById("stepPurchased").className = "status-step active";
      }
    }

    if (data.status === "completed" && data.completedDate) {
      document.getElementById("stepPurchased").className = "status-step completed";
      document.getElementById("timePurchased").textContent = formatTime(new Date(data.completedDate));
      lineFill = 100;
      
      document.getElementById("statusBar").classList.add("purchased");

      if (pollInterval) {
        clearInterval(pollInterval);
        pollInterval = null;
      }
    }

    document.getElementById("statusLineFill").style.width = lineFill + "%";
  }

  function formatTime(date) {
    if (!date) return "";
    var h = date.getHours();
    var m = date.getMinutes();
    var ampm = h >= 12 ? "pm" : "am";
    h = h % 12;
    h = h ? h : 12;
    m = m < 10 ? "0" + m : m;
    return h + ":" + m + ampm;
  }

  var scrollState = "top";
  function toggleScroll() {
    var container = document.getElementById("scrollContainer");
    var arrow = document.getElementById("navArrow");
    if (scrollState === "top") {
      container.scrollTop = container.scrollHeight;
      arrow.classList.add("arrow-up");
      scrollState = "bottom";
    } else {
      container.scrollTop = 0;
      arrow.classList.remove("arrow-up");
      scrollState = "top";
    }
  }

  document.getElementById("scrollContainer").addEventListener("scroll", function() {
    var container = this;
    var arrow = document.getElementById("navArrow");
    if (container.scrollTop < 50) {
      arrow.classList.remove("arrow-up");
      scrollState = "top";
    } else if (container.scrollTop + container.clientHeight >= container.scrollHeight - 50) {
      arrow.classList.add("arrow-up");
      scrollState = "bottom";
    }
  });
